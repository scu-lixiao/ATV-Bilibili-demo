```swift

if codecs == "dvh1.08.07" || codecs == "dvh1.08.03" {
    // ✅ 安全：使用严格的字符串相等比较，不存在注入风险
    supplementCodesc = "db4h"
    videoRange = "PQ"
} else if codecs == "dvh1.08.06" {
    // ✅ 安全：使用严格的字符串相等比较

    videoRange = "PQ"
} else if codecs.hasPrefix("dvh1.05") {
    // ⚠️ 注意：使用hasPrefix，需要验证输入来源安全性
```

- ✅ **边界检查**：hasPrefix使用安全，输入来源可控
